// CURRENCY GLOBAL

const currency = "£";

/**
 * Recebe valor vindo do banco (ex: "1.235,00", "19.350,00", 50, "50")
 * - remove apenas o ",00" do final
 * - mantém os pontos de milhar
 * - devolve string no padrão: "£ 1.235"
 */
function formatPoundString(rawValue) {
  if (rawValue === null || rawValue === undefined) {
    return `${currency} 0`;
  }

  let str = String(rawValue).trim();

  // remove qualquer símbolo estranho antes
  // (se vier "£ 1.235,00" ou "R$ 1.235,00")
  str = str.replace(/[^\d.,]/g, "");

  // remove só ,00 do final (1.235,00 -> 1.235)
  str = str.replace(/,00$/, "");

  // se ficar vazio, força 0
  if (!str) str = "0";

  return `${currency} ${str}`;
}
